//8 directions per 1 frame

if (motion.x != 0 || motion.y != 0)
{

sprite_moving = PolishR
anim_dir = round(point_direction(x,y,x+motion.x,y+motion.y))
var car_dir = round(anim_dir/45)
var _total_frame = sprite_get_number(sprite_moving)/4
localFrame += sprite_get_speed(sprite_moving)/60
image_moving = car_dir + (floor(localFrame)*_total_frame)
if (localFrame >= _total_frame)
{
	localFrame = 0
}

}
else
{
	sprite_moving = PolishR
	var car_dir = round(anim_dir/45)
	var _total_frame = sprite_get_number(sprite_moving)/1
	//localFrame += sprite_get_speed(sprite_moving)/60
	image_moving = car_dir + (floor(localFrame)*_total_frame)
	if (localFrame >= _total_frame)
	{
		localFrame = 0
	}
}
